#load topic models library
library(topicmodels)

#Set parameters for Gibbs sampling
burnin <- 10000
iter <- 20000
thin <- 500
seed <-list(1,23,4,5,67)
 
nstart <- 5
best <- TRUE

#Number of topics
T <- 4



#Run LDA using Gibbs sampling
ldaOut <-LDA(dtm,T, method="Gibbs", control=list(nstart=nstart, seed = seed, best=best, burnin = burnin, iter = iter, thin=thin))

#probabilities associated with each topic assignment
topicProbabilities <- as.data.frame(ldaOut@gamma)
