myreg = function(y,X) {
#purpose :compute least square regression
#arguement : y-vector of dep var, X- array of indep vars
#output : list containing lsq coefficients and standard errors

bhat=chol2inv(chol(t(X)%*%X))%*%t(X)%*% y
 yhat=X%*%bhat 
 resid=y-yhat
 ybar=mean(y)
 SST=sum((y-ybar)^2)
 SSE=sum((y-yhat)^2)
 SSR=sum((yhat-ybar)^2)
 Rsq=SSR/SST
 sigma=SSE/(length(y)-ncol(X))
 stderror=sqrt(sigma) 
 covb <- sigma*chol2inv(chol(t(X)%*%X))
 seb = sqrt(diag(covb))
 tstat=bhat/seb
 list(bhat=bhat, yhat=yhat, resid=resid, ybar=ybar, SST=SST, SSE=SSE,  SSR=SSR, Rsq=Rsq, stderror=stderror, seb=seb, tstat=tstat)
}

## example of processing for use with rhierLinearModel
 
  retailer = levels(cheese$RETAILER)
  nreg = 10
  nvar = 3
  regdata = NULL
  for (reg in 1:nreg) {
    y = log(cheese$VOLUME[cheese$RETAILER==retailer[reg]])
    iota = c(rep(1,length(y)))
    X = cbind( iota, cheese$DISP[cheese$RETAILER==retailer[reg]],
      cheese$PRICE[cheese$RETAILER==retailer[reg]])
    regdata[[reg]] = list(y=y, X=X)
  }
  Z = matrix(c(rep(1,nreg)), ncol=1)
  nz = ncol(Z)
  
  
  ## run each individual regression and store results
  lscoef = matrix(double(nreg*nvar), ncol=nvar)
  
  for (reg in 1:nreg) {
    coef = lsfit(regdata[[reg]]$X, regdata[[reg]]$y, intercept=FALSE)$coef
    if (var(regdata[[reg]]$X[,2])==0) {
      lscoef[reg,1]=coef[1] 
      lscoef[reg,3]=coef[2]
    }
    else {lscoef[reg,]=coef}
  }
